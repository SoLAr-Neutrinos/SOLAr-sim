# SLArMCPrimaryInfo dictionary

  ${PROJECT_SOURCE_DIR}/include/event/SLArMCPrimaryInfo.hh

  LINKDEF
  ${PROJECT_SOURCE_DIR}/include/event/SLArMCPrimaryInfoLinkDef.h)

  )
